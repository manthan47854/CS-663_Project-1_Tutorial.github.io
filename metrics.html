Angles from 2D

function angleDeg(a,b,c){
  const ab = {x:a.x-b.x, y:a.y-b.y};
  const cb = {x:c.x-b.x, y:c.y-b.y};
  const dot = ab.x*cb.x + ab.y*cb.y;
  const mab = Math.hypot(ab.x,ab.y); const mcb = Math.hypot(cb.x,cb.y);
  return Math.acos(Math.max(-1,Math.min(1, dot/(mab*mcb))))*180/Math.PI;
}

Symmetry index: 100 * |L - R| / (0.5*(L + R) + ε).
